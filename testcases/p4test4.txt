global arr;
array arr[1..10];
global len = 10;

arr[1] = 123;
arr[2] = 23;
arr[3] = 1234;
arr[4] = 1;
arr[5] = 21;
arr[6] = 11;
arr[7] = 213;
arr[8] = 22;
arr[9] = 121;

foreach i in 1..len do
    foreach j in 1..len-i do
        if arr[j]>arr[j+1] then
            arr[j] <-> arr[j+1];
        end if
    end for
end for
